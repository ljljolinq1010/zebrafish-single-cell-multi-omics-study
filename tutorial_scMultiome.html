## conduct a correlation analysis for a gene of interest to identify peaks whose chromatin accessibility is associated with gene expression throughout the developmental trajectory
DefaultAssay(noto.trj)<-"ATAC"
noto.trj <- RegionStats(noto.trj, genome = BSgenome.Drerio.UCSC.danRer11)
genetoplpt<-"tbxta" 
noto.trj <- LinkPeaks( 
  object = noto.trj,
  distance = 50000,
  method = "pearson",
  genes.use = genetoplpt,
  peak.assay = "ATAC",
  expression.assay = "SCT",
  pvalue_cutoff = 0.05,
  score_cutoff = 0.05
)

Motif entichment analysis for cell type specific peaks

for more details, please refer to: https://stuartlab.org/signac/articles/motif_vignette

## add motif information to the Seurat object
noto.trj <- AddMotifs(
  object = noto.trj,
  genome = BSgenome.Drerio.UCSC.danRer11,
  pfm = zf.motifs
)

## identify differentially accessible peaks
da_peaks <- FindMarkers(
  object = noto.trj,
  ident.1 = '8-notochord(bud)',
  only.pos = TRUE,
  test.use = 'LR',
  min.pct = 0.05,
  latent.vars = 'nFeature_ATAC'
)

## get top cell type specific peaks 
top.da.peak <- da_peaks[(da_peaks$p_val_adj < 0.01 & da_peaks$avg_log2FC > 2 & da_peaks$pct.1>0.1),]
